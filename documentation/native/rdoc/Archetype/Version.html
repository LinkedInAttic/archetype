    version() click to toggle source

    Returns a string representing the version.

    The :major, :minor, and :teeny keys have their respective numbers. The :string key contains a human-readable string representation of the version. The :rev key will have the current revision hash.

    Method borrowed from Compass. All credit goes to Chris Eppstein and other contributors github.com/chriseppstein/compass/blob/stable/lib/compass/version.rb (This method swiped from Haml and then modified, some credit goes to Nathan Weizenbaum)

  else

  end

def read_version
  require 'yaml'
  @version = YAML::load(File.read(scope('VERSION.yml')))
  @version[:teeny] = @version[:patch]
  @version[:string] = "#{@version[:major]}.#{@version[:minor]}"
  @version[:string] << ".#{@version[:patch]}" if @version[:patch]
  @version[:string] << ".#{@version[:build]}" if @version[:build]
  @version[:string] << ".#{@version[:state]}" if @version[:state]
  @version[:string] << ".#{@version[:iteration]}" if @version[:iteration]
  if !ENV['OFFICIAL'] && r = revision
    @version[:string] << ".#{r[0..6]}"
    @version[:rev] = r
  end
  @version
end

def revision_from_git
  if File.exists?(scope('.git/HEAD'))
    Dir.chdir scope(".") do
      %xgit rev-parse HEAD`
    end
  end
end
